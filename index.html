PFA Evidență Contabilă

Această aplicație salvează toate datele tale direct în Google Sheets. Urmează pașii de mai jos o singură dată, apoi intri cu URL-ul tău personal.

PAS 1
Creează un Google Spreadsheet nou
Mergi la sheets.google.com → creează un spreadsheet nou → dă-i un nume (ex. "PFA Evidență 2024").
PAS 2
Deschide Apps Script
În spreadsheet: Extensii → Apps Script. Șterge tot codul existent și lipește codul de mai jos:

const ss = SpreadsheetApp.getActiveSpreadsheet(); const result = handleRequest(ss, data); return ContentService.createTextOutput(JSON.stringify({ok:true, data:result})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action || 'getAll'; const sheet = e.parameter.sheet; const callback = e.parameter.callback; let result = {}; if (action === 'getAll') { Object.entries(SHEETS).forEach(([key, name]) => { const sh = getOrCreateSheet(ss, name); result[key] = sh.getDataRange().getValues(); }); } else if (action === 'getSheet' && sheet) { const sh = getOrCreateSheet(ss, SHEETS[sheet] || sheet); result = sh.getDataRange().getValues(); } const json = JSON.stringify({ok:true, data:result}); if (callback) { return ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } function handleRequest(ss, data) { const {action, sheet, row, id} = data; const sheetName = SHEETS[sheet] || sheet; const sh = getOrCreateSheet(ss, sheetName); if (action === 'append') { sh.appendRow(row); return {appended: true}; } else if (action === 'delete') { const rows = sh.getDataRange().getValues(); for (let i = rows.length - 1; i >= 0; i--) { if (String(rows[i][0]) === String(id)) { sh.deleteRow(i + 1); return {deleted: true}; } } return {deleted: false}; } else if (action === 'update') { const rows = sh.getDataRange().getValues(); for (let i = 0; i < rows.length; i++) { if (String(rows[i][0]) === String(id)) { row.forEach((val, j) => sh.getRange(i + 1, j + 1).setValue(val)); return {updated: true}; } } return {updated: false}; } else if (action === 'initSheets') { initHeaders(ss); return {initialized: true}; } return {}; } function getOrCreateSheet(ss, name) { return ss.getSheetByName(name) || ss.insertSheet(name); } function initHeaders(ss) { const headers = { "Incasari": ["ID","Data","Categorie","Descriere","Client","Suma","Moneda","NrFact"], "Cheltuieli": ["ID","Data","Categorie","TipDed","ProcDed","Descriere","Furnizor","SumaBruta","SumaDeductibila","NrDoc"], "MijloaceFixe": ["ID","Data","Denumire","Categorie","ValoareAchiz","DurataAmort","AmortizareLunara","DataPIF","NumarInv","Observatii"], "ObiectInv": ["ID","Data","Denumire","Cantitate","ValoareUnit","ValoareTotala","Locatie","Stare"], "Contracte": ["ID","DataInceput","DataSfarsit","Partener","Tip","Valoare","Moneda","Descriere","Stare"], "REF": ["ID","An","TotalVenituri","TotalCheltuieliDed","TotalCheltuieliPartial","TotalCheltuieliNeded","VenitNet","CAS","CASS","Impozit","TotalTaxe","Note"] }; Object.entries(headers).forEach(([name, hdrs]) => { const sh = getOrCreateSheet(ss, name); if (sh.getLastRow() === 0) sh.appendRow(hdrs); }); }

Despre aplicație

PFA Evidență Contabilă v2.0 — aplicație pentru activități de asigurări, turism, leasing și credite.
Datele sunt stocate exclusiv în Google Sheets-ul tău.
Registre incluse: RJIP, REF, Mijloace Fixe, Obiecte Inventar, Contracte.
Nu se transmite niciun fel de date către terți.
